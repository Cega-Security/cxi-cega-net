Detailed Description

Clase principal que permite la conexión al servicio, ya sea a través de un simulador, un HSM, o un servicio en la nube. Esta clase permite realizar operaciones criptográficas
ya sean firmas digitales o encripción de información, también permite la verificación de una firma así como la manipulación de las llaves criptográficas.

public string Ejemplo(string datos, string nombre, string grupo)
{
CxiCega.Cxi cxi = new CxiCega.Cxi("192.168.1.1", 10000, 60000);
cxi.Logon("cxi","123456");

byte[] data = System.Text.Encoding.UTF8.GetBytes(datos);
int HashAlgo = CxiCega.Cxi.MECH_HASH_ALGO_SHA1;

var hashDigest = cxi.createHash(HashAlgo, datos);

CxiCega.MechParam m = new CxiCega.MechParam(HashAlgo | CxiCega.Cxi.MECH_PAD_PKCS1);
var keyTemplate = new CxiCega.KeyProperties(nombre, grupo);
var key = cxi.findKey(0, keyTemplate);
byte[] dataSigned = cxi.sign(0, key, m, hashDigest);

cxi.Close();
return Convert.ToBase64String(dataSigned);
}
